//Time:O(n)
//Space:O(n)
class Solution {
    public void moveZeroes(int[] nums) {
        int index = 0;
        for (int i=0; i<nums.length; i++){
            if (nums[i]!=0){
                int tmp = nums[index];
                nums[index] = nums[i];
                nums[i] = tmp;
                index++;
            }
        }
    }
}
